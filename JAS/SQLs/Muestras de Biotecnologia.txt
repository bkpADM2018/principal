Select HC.DTCONTABLE, HCD.NUCARTAPORTE, MUESTRA from 
HCAMIONESDESCARGA HCD 
inner join HCAMIONES HC on HC.DTCONTABLE=HCD.DTCONTABLE and HC.IDCAMION=HCD.IDCAMION
left join TBLMUESTRASBIOTECNOLOGIA MB on HCD.NUCARTAPORTE=MB.NUCARTAPORTE
where 
HCD.DTCONTABLE>='2015-06-01' and HCD.DTCONTABLE<='2015-06-30'
AND HC.CDPRODUCTO=23
and HC.CDESTADO in (6, 8)
